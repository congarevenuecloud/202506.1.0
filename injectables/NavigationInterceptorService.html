Description

Service to intercept and manage navigation URLs by prepending a base path to the current URL based on the router param map passed. It ensures consistent routing based on the configuration and updates navigation dynamically.

import { NavigationInterceptorService, AObjectService } from '@congarevenuecloud/ecommerce';

export class MyComponent implements OnInit {
constructor( private navigationInterceptorService: NavigationInterceptorService)
}
// or
export class MyService extends AObjectService {
private navigationInterceptorService: NavigationInterceptorService = this.injector.get(NavigationInterceptorService);
}
